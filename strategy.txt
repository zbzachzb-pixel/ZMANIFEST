🚀 Load Management Timer & Scheduling System - Implementation Plan
📋 Context & Requirements
Current System State

Load management system with drag-and-drop functionality
Load statuses: building → ready → departed → completed
Instructors can be assigned to students in loads
Students can be dragged from queue to loads without instructors first
Visual indicators for unassigned students

New Feature: Countdown Timer & Scheduling System
Business Logic:

40-minute instructor cycle: 20 min prep + 20 min skydive/debrief
Pattern: Instructors available for loads 1, 3, 5, 7 (every other load)
Countdown starts: When Load 1 marked "Ready" → 20-min countdown begins
Cascade: When Load 1 marked "Departed" → Load 2's countdown starts immediately (regardless of Load 2's status)
Delay feature: Push all subsequent loads back by X minutes (configurable, default 20)

Step 4: Countdown Timer Logic

Create custom hook for countdown management:
typescriptimport { useState, useEffect } from 'react'
import type { Load } from '@/types'

interface CountdownState {
  timeRemaining: number | null  // Seconds remaining
  status: 'waiting' | 'countdown' | 'ready' | 'departed'
  displayText: string
}

export function useLoadCountdown(
  load: Load,
  allLoads: Load[],
  minutesBetweenLoads: number
): CountdownState {
  const [timeRemaining, setTimeRemaining] = useState<number | null>(null)
  
  useEffect(() => {
    // If load is departed or completed, no countdown
    if (load.status === 'departed' || load.status === 'completed') {
      setTimeRemaining(null)
      return
    }
    
    // Find previous load in sequence
    const previousLoad = allLoads.find(l => l.position === load.position - 1)
    
    // COUNTDOWN STARTING CONDITIONS:
    // 1. This load is "Ready" AND it's Load #1 (no previous load)
    // 2. Previous load is "Departed" (regardless of this load's status)
    
    const shouldStartCountdown = 
      (load.status === 'ready' && !previousLoad) ||
      (previousLoad && previousLoad.status === 'departed')
    
    if (!shouldStartCountdown) {
      setTimeRemaining(null)
      return
    }
    
    // Start or continue countdown
    if (!load.countdownStartTime) {
      // Initialize countdown start time
      // This should trigger a database update
      const startTime = new Date().toISOString()
      // You'll need to expose an update function here
      // For now, just calculate based on now
    }
    
    const interval = setInterval(() => {
      if (load.countdownStartTime) {
        const startTime = new Date(load.countdownStartTime).getTime()
        const now = Date.now()
        const elapsed = Math.floor((now - startTime) / 1000) // seconds
        const totalSeconds = minutesBetweenLoads * 60
        const remaining = totalSeconds - elapsed
        
        setTimeRemaining(Math.max(0, remaining))
      }
    }, 1000)
    
    return () => clearInterval(interval)
  }, [load, allLoads, minutesBetweenLoads])
  
  // Determine status and display text
  const getStatus = (): CountdownState => {
    if (load.status === 'departed') {
      return { timeRemaining: null, status: 'departed', displayText: '✈️ In Air' }
    }
    
    if (load.status === 'completed') {
      return { timeRemaining: null, status: 'departed', displayText: '🎉 Completed' }
    }
    
    const previousLoad = allLoads.find(l => l.position === load.position - 1)
    
    if (load.status !== 'ready') {
      if (previousLoad && previousLoad.status !== 'departed') {
        return { timeRemaining: null, status: 'waiting', displayText: '⏸️ Building' }
      }
      return { timeRemaining: null, status: 'waiting', displayText: '🔨 Building' }
    }
    
    // Load is Ready
    if (previousLoad && previousLoad.status !== 'departed') {
      return { 
        timeRemaining: null, 
        status: 'waiting', 
        displayText: `⏸️ Waiting for Load #${previousLoad.position}` 
      }
    }
    
    // Countdown active
    if (timeRemaining !== null) {
      if (timeRemaining === 0) {
        return { timeRemaining: 0, status: 'ready', displayText: '✅ Ready to Depart' }
      }
      
      const minutes = Math.floor(timeRemaining / 60)
      const seconds = timeRemaining % 60
      return { 
        timeRemaining, 
        status: 'countdown', 
        displayText: `⏱️ ${minutes}:${seconds.toString().padStart(2, '0')}` 
      }
    }
    
    return { timeRemaining: null, status: 'waiting', displayText: '⏸️ Ready' }
  }
  
  return getStatus()
}

// Modify status change handler to trigger countdowns
const handleStatusChange = async (loadId: string, newStatus: Load['status']) => {
  const load = loads.find(l => l.id === loadId)
  if (!load) return
  
  try {
    await updateLoad(loadId, { status: newStatus })
    
    // If marking as Ready and it's Load #1, start countdown
    if (newStatus === 'ready' && load.position === 1) {
      await startLoadCountdown(loadId)
    }
    
    // If marking as Departed, start next load's countdown
    if (newStatus === 'departed') {
      const nextLoad = loads.find(l => l.position === load.position + 1)
      if (nextLoad && !nextLoad.countdownStartTime) {
        await startLoadCountdown(nextLoad.id)
      }
    }
  } catch (error) {
    console.error('Failed to update status:', error)
  }
}

typescriptconst handleDelayLoad = async (loadId: string) => {
  const load = loads.find(l => l.id === loadId)
  if (!load) return
  
  if (load.status !== 'ready') {
    alert('⚠️ Can only delay loads in Ready status')
    return
  }
  
  const delayMinutes = loadSettings.minutesBetweenLoads
  
  if (!confirm(`Delay Load #${load.position} and all subsequent loads by ${delayMinutes} minutes?`)) {
    return
  }
  
  try {
    // Update this load
    await updateLoad(loadId, {
      status: 'building',
      countdownStartTime: null,
      delayMinutes: (load.delayMinutes || 0) + delayMinutes
    })
    
    // Update all subsequent loads
    const subsequentLoads = loads.filter(l => l.position > load.position)
    for (const subLoad of subsequentLoads) {
      await updateLoad(subLoad.id, {
        delayMinutes: (subLoad.delayMinutes || 0) + delayMinutes,
        countdownStartTime: null // Reset countdowns
      })
    }
    
    // Auto-optimize all building loads
    console.log('🔄 Re-optimizing loads after delay...')
    await executeOptimization()
    
  } catch (error) {
    console.error('Failed to delay load:', error)
    alert('Failed to delay load')
  }
}

typescriptconst findBestInstructor = (
  student: QueueStudent,
  targetLoad: Load,
  allLoads: Load[],
  existingAssignments: LoadAssignment[],
  usedInstructors: Set<string>
): Instructor | null => {
  
  const qualified = instructors.filter(i => {
    if (!i.clockedIn) return false
    if (usedInstructors.has(i.id)) return false
    
    // Check qualification
    if (student.jumpType === 'tandem' && !i.tandem) return false
    if (student.jumpType === 'aff' && !i.aff) return false
    
    // NEW: Check if instructor has enough turnaround time
    // Find instructor's last assigned load
    const instructorLoads = allLoads
      .filter(l => (l.assignments || []).some(a => a.instructorId === i.id))
      .sort((a, b) => a.position - b.position)
    
    if (instructorLoads.length > 0) {
      const lastLoad = instructorLoads[instructorLoads.length - 1]
      const loadGap = targetLoad.position - lastLoad.position
      const instructorCycleLoads = Math.ceil(loadSettings.instructorCycleTime / loadSettings.minutesBetweenLoads)
      
      if (loadGap < instructorCycleLoads) {
        console.log(`❌ ${i.name} not available: Last on Load #${lastLoad.position}, needs ${instructorCycleLoads} load gap, only ${loadGap} loads apart`)
        return false
      }
    }
    
    return true
  })
  
  if (qualified.length === 0) return null
  
  // Sort by balance (lowest first)
  qualified.sort((a, b) => {
    const balanceA = instructorBalances.get(a.id) || 0
    const balanceB = instructorBalances.get(b.id) || 0

Instructor Availability

Settings

 Change instructor cycle time → affects availability calculations
 Settings persist after page refresh

Edge Cases

 Delete a load in the middle → positions renumber correctly
 Mark load Ready out of order → correct countdown behavior
 Multiple delays on same load → cumulative delay tracked
 Browser tab inactive → countdown syncs on return
